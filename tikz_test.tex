\documentclass{article}

\usepackage{tikz}

\begin{document}


\begin{tikzpicture}

\def \n {5}
\def \radius {3cm}
\def \margin {8}

\foreach \s in {1,...,\n}
{
    \node[draw, circle] at ({360/\n * (\s - 1)}:\radius) {$\s$};
    \draw[->, >=latex] ({360/\n * (\s - 1) + \margin}:\radius)
        arc ({360/\n * (\s - 1) + \margin}:{360/\n * (\s)- \margin}:\radius)
}

\end{tikzpicture}


\end{document}
